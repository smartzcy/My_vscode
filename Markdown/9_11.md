```c
            return 1;

            return -2;

        if (day <= flag + 28 && day >= 1)
            return 1;

            return -2;
    }
```

```cpp
#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
using namespace std;
const int INF = 1e9;
const int N = 1e6 + 10;
int n, m;
int a[N], b[N];
int main() {
    int T;
    cin >> T;
    while (T--) {
        cin >> n;
        int maxx = 0;
        for (int i = 1; i <= n; i++) {
            cin >> a[i];
            b[i] = 0;
            if (a[i] > maxx) b[i] = 1;
            maxx = max(a[i], maxx);
        }
        int minn = INF;
        for (int i = n; i >= 1; i--) {
            if (a[i] < minn) b[i]++;
            minn = min(a[i], minn);
        }
        vector<int> vec;
        for (int i = 1; i <= n; i++) {
            if (b[i] == 2) vec.push_back(a[i]);
        }
        cout << vec.size() << endl;
        for (auto i : vec) cout << i << " ";
        cout << endl;
    }
}
```

```cpp
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
#include <map>
using namespace std;
const int INF = 1e9;
const int N = 1e6 + 10;
int n, m;
int gas[N], cost[N];
int check() {
    int sum = 0, i = 0;
    bool vis[N];
    memset(vis, false, sizeof(vis));
    while (i < n && !vis[i]) {
        vis[i] = true;
        sum = gas[i];
        int pos = i % n;
        while (sum - cost[pos] >= 0) {
            sum += gas[(pos + 1) % n] - cost[pos];
            if (sum < 0) break;
            pos = (pos + 1) % n;
            if (pos == i) return pos;
        }
        i = pos + 1;
    }
    return -1;
}
int main() {
    cin >> n;
    for (int i = 0; i < n; i++) cin >> gas[i];
    for (int i = 0; i < n; i++) cin >> cost[i];
    cout << check();
}
```
